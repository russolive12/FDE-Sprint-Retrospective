No Supabase yet? Follow the 4 steps below — free, no credit card.
1

2
Click New project. Name it (e.g. retro-board). Pick region. Set a password. Click Create. Wait ~1 min.
3
Go to SQL Editor → paste and run this: create table notes ( id uuid primary key default gen_random_uuid(), text text not null, theme text not null, subcat text default '', author text default '', created_at timestamptz default now() ); create table votes ( id uuid primary key default gen_random_uuid(), note_id uuid references notes(id) on delete cascade, voter_id text not null, unique(note_id, voter_id) ); alter table notes enable row level security; alter table votes enable row level security; create policy "r" on notes for select using (true); create policy "i" on notes for insert with check (true); create policy "d" on notes for delete using (true); create policy "u" on notes for update using (true); create policy "rv" on votes for select using (true); create policy "iv" on votes for insert with check (true); create policy "dv" on votes for delete using (true);
